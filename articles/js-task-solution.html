Простенький пример того, как добавить комментарий с помощью js. Первый пример написан на чистом js:

HTML структура:

    < body>
       < div class="wrapper">
           < div class="container-coments">< /div>
           < div class="container-write-coments">
               < p>Напишите тут ваш комментарий:< /p>
               < p>Ваш ник: < input type="text" class="nikname">< /p>
               < p>Коментарий:< /p>
               < textarea cols="90" rows="8" class="comment">< /textarea>< br>< br>
               < button class="add-comment">Добавить< /button>
           < /div>
       < /div>
    < /body>

JS:

    'use strict';

    let containerComents = document.querySelector('.container-coments');
    let nikname = document.querySelector('.nikname');
    let comment = document.querySelector('.comment');
    let btnAddComment = document.querySelector('.add-comment');


    btnAddComment.addEventListener('click', handler);

    function handler(){
        let divCom = document.createElement('div');
        let img = document.createElement('img');
        let div = document.createElement('div');
        let div2 = document.createElement('div');
        let p = document.createElement('p');
        let nik = document.createElement('p');

        divCom.className = 'd';
        img.src = 'img/ava.jpg';
        p.innerHTML = comment.value;
        nik.innerHTML = nikname.value;

        div2.append(img, nik);
        div.append(p);
        divCom.append(div2, div);
        containerComents.append(divCom);
    }

Задача 1

Файл с кодом и стилями находится по пути articles/iframe/task2/task2.html

Нужно по клику на маленькой картинке показать её большую копию.

HTML:

        < div class="wrapper">
           < div class="lg">
               < img src="img/img1-lg.jpg">
           < /div>
           < div class="small">
               < img src="img/img1.jpg" data-num="1">
               < img src="img/img2.jpg" data-num="2">
               < img src="img/img3.jpg" data-num="3">
               < img src="img/img4.jpg" data-num="4">
               < img src="img/img5.jpg" data-num="5">
           < /div>
       < /div>

JS:

    'use strict';

    let small = document.querySelector('.small');
    let lg = document.querySelector('.lg');
    small.addEventListener('click', handler);

    function handler(e){
        let img = e.target;
        if(img.tagName != 'IMG') return;

        let num = img.dataset.num;


        let str = `img/img${num}-lg.jpg`;
        let pic = lg.querySelector('img');
        pic.src = str;
    }

jQuery:

    $('.small').on('click', function(e){
        let img = e.target;
        if(img.tagName != 'IMG') return;

        let num = $(img).data('num');

        let str = `img/img${num}-lg.jpg`;
        $('.lg img').attr('src', str);
    })

            target.classList.remove('color');
            return;

        if(e.ctrlKey || e.metaKey){
            target.classList.add('color');
        }else{
            for(let val of list.children){
                val.classList.remove('color');
            }
            target.classList.add('color');
        }
    }

Задача 3

Файл с кодом и стилями находится по пути articles/iframe/task4/task4.html

Задача аналогичная задаче выше, но добавляется веделение промедутка между лишками при зажатом Shift.

HTML:

        < ul class="list" onmousedown="return false">
            < li data-list="0">Текст< /li>
            < li data-list="1">Текст< /li>
            < li data-list="2">Текст< /li>
            < li data-list="3">Текст< /li>
            < li data-list="4">Текст< /li>
            < li data-list="5">Текст< /li>
            < li data-list="6">Текст< /li>
            < li data-list="7">Текст< /li>
            < li data-list="8">Текст< /li>
            < li data-list="9">Текст< /li>
        < /ul>

JS:

    let list = document.querySelector('.list');
    let lis = document.querySelectorAll('.list li');


    list.onclick = function (event){
        let target = event.target;
        if(target.tagName != 'LI') return;

        if(event.shiftKey == false){//если shift не нажат выполняется этот блок кода

            if(target.classList.contains('color')){
                target.classList.remove('color');
                return;
            }

            if(event.ctrlKey || event.metaKey){
                target.classList.add('color');
            }else{
                for(let val of list.children){
                    val.classList.remove('color');
                }
                target.classList.add('color');
            }
        }else{// а если shift нажат, то выполняется этот блок
            let res = startCheck();

            let number = target.dataset.list;
            if(number > res){//этот блок закрашивает в фон сверху вниз
                for(let i = res[0]; i <= number; i++){
                    lis[i].classList.add('color');
                }
            }else{
                for(let i = number; i <= res[res.length - 1]; i++){// а этот блок закрашивает снизу вверх
                    lis[i].classList.add('color');
                }
            }		
        }
    }

    function startCheck (){//проверочная функция, которая в массив добавляет индексы закрашенных элементов li
        let mas = [];
        for (let i = 0; i < lis.length; i++) {
            if(lis[i].classList.contains('color')){
                mas.push(i);
            }
        }
        return mas;
    }

Задача 4

Файл с кодом и стилями находится по пути articles/iframe/task5/task5.html

При наведении мыши на элемент-кнопку с атрибутом data-tooltip, над ним должна показываться подсказка и скрываться при переходе на другой элемент. Подсказка не должна вылезать за края окна, если она вылезает вверх, то нужно вывести ниже, если вылезает влево, то дальше левой границы тоже не должна вылезать.

HTML:

        < button data-tooltip="Это очень длинная подсказка">Кнопка1< /button>
        < button data-tooltip="HTML< br>подсказка">Кнопка2< /button>

JS:

        document.addEventListener('mouseover', handler);
        document.addEventListener('mouseout', handler2);

        function handler(e){
            let target = e.target;
            if(target.tagName != 'BUTTON') return;

            let text = target.dataset.tooltip;
            let btnCoords = target.getBoundingClientRect();

            let tooltip = createTooltip(text);
            document.body.append(tooltip);

            let t = btnCoords.top  - tooltip.offsetHeight - 8;
            let l = btnCoords.left + (target.offsetWidth / 2) - (tooltip.offsetWidth / 2);
            if(t < 0){
                tooltip.style.top = target.offsetHeight + btnCoords.top + 8 + 'px';
            }else{
                tooltip.style.top = t + 'px';
            }
            if(l < 0){
                tooltip.style.left = 2 + 'px';
            }else{
                tooltip.style.left = l + 'px';
            }
        }

        function createTooltip(text){
            let elem = document.createElement('span');
            elem.classList.add('tooltip');
            let msg = document.createTextNode(text);
            elem.insertAdjacentHTML('afterbegin', msg.nodeValue);
            return elem;
        }

        function handler2(){
            let elem = document.querySelector('.tooltip');
            if(elem){
                elem.remove();
            }
        }
