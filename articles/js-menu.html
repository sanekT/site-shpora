Создание 3х уровнего меню на JS и CSS

Будем делать меню похожее на то, что находится на сайте ozon.ru. Там есть кнопка "Каталог", при нажатии на котоую открывается это меню. Меню состоит из контейнера и в нём есть левое меню и правое меню. При наведении на пункты в левом меню открываются подпункты в право меню. В правом меню подпункты расположены по колоночной вёрстке. Как верстать колоночной вёрсткой Денис показывает с 10й минуты в этом видео - тут

Мы будем эффект hover делать не на css, а на js. Вообще как бы событие hover в чистом js не существует. В библиотеках типо jQuery это событие сделано. Поэтому в js мы будем событие hover делать через другие события, которые есть, это 'mouseover', 'mouseout', 'mouseenter', 'mouseleave'. Первые 2 делают тоже что и последние 2, только есть небольшая разница. Последние 2 более конкретные события. При срабатывании первых 2х событий происходит всплытие. А события 'mouseenter', 'mouseleave' не всплывают и если повесить эти события на блок родитель, то при наведении на дочернии элементы, событие будет показывать только родителя, т.е. e.target будет показывать только родителя. Использовать эти события нужно исходя из того что нам нужно. Мы будем использовать первые 2 события со всплытиями.

                    <script>
                        //этот объект имитирует эффект hover
                        const Hover = {
                            menu: {},
                            subMenu: {},
                            currentElement: null,
                            timeoutOver: null,
                            time: 100,

                            activate: function(menu, subMenu, time){

                                if(typeof menu === 'undefined' || typeof subMenu === 'undefined'){
                                    return;
                                }
                                if(typeof time !== 'undefined') this.time = parseInt(time);

                                this.menu = document.querySelector(menu);

                                this.subMenu = document.querySelector(subMenu);

                                //вешаем события
                                this.over();
                                this.out();
                            },

                            over: function(){
                                /*
                                    вот тут хорошо видно разницу между обычной функцией и стрелочной и работу this внутри этих функций. Столкнулся с такой ситуацией, мы вешаем событие на свойство this.menu и присваиваем ей обычную функцию:
                                    this.menu.onmouseover = function(){
                                        //и внутри функции как мне обратиться к свойству this.menu?
                                        //если отсюда я напишу this.menu, то внутри this будет объект перед
                                        //точкой, а имеено  в this будет this.menu и поэтому получается если 
                                        //изнутри вызвать this.menu, то обращение будет выглядеть 
                                        //this.menu.menu, поэтому я и присваиваю свойства в переменные, и 
                                        //внутри уже обращаюсь к этим переменным, в которые скопированы 
                                        //свойства, но тогда если я хочу менять свойства внутри этой функции
                                        //они не поменяются, а поменяются только переменные. Поэтому тут 
                                        //проще использовать в виде обработчика события стрелочную функцию
                                    }
                                */
                                this.menu.onmouseover = e => {
                                    clearTimeout(this.timeoutOver);

                                    /*
                                        помещаем весь код, котрый срабатывает при наведении, в функцию 
                                        setTimeout(), чтобы притормозить отклик при наведении на элемент
                                    */
                                    this.timeoutOver = setTimeout(() => {
                                        if((this.menu.contains(e.relatedTarget) &&  this.currentElement) || this.menu === e.target) return;

                                        let parentElement = e.target;

                                        //получаем все пункты меню в HTMLCollection и через деструктивное присваивание раскидаем пункты меню в массив. В menuItems будет настоящий массив с пунктами, а не псевдомассив
                                        let menuItems = [...this.menu.children];
                                        //то же самое сделаем и с правой частью меню
                                        let subMenuItems = [...this.subMenu.children];

                                        menuItems.forEach(index => index.removeAttribute('style'));
                                        subMenuItems.forEach(index => index.removeAttribute('style'));

                                        /*
                                            этот цикл делает следующее. У нас пункт меню лежит в родительском элементе div, а внутри лежит текст название пункта, а этот текст например может быть обёрнут в span. Мы хотим добиться эффекта, когда при наведении на пункт меню, у нас в target был всё время родитель div и никак не  показывал бы нам span. Такого же состояния можно дабиться если в цикле получить все пункты меню и повесить на них событие mouseenter, это событие будет только показывать div родитель и пропускать наведение на внутренние элементы. Надо попробывать такой вариант.
                                        */
                                        while(this.menu !== parentElement && this.menu.contains(parentElement)){
                                            this.currentElement = parentElement;
                                            parentElement = this.currentElement.parentNode;
                                        }

                                        if(!this.currentElement) return;

                                        //ищем на каком пункте div мы находимся
                                        let index = menuItems.indexOf(this.currentElement);
                                        this.currentElement.style.backgroundColor = '#f3f0f0';

                                        //и теперь надо показать подменю при наведении на пункт меню
                                        subMenuItems[index].style.display = 'block';
                                    }, this.time);

                                };
                            },

                            out: function(){
                                this.menu.onmouseout = e => {
                                    //console.log(e.relatedTarget);
                                    if(!this.currentElement || this.menu === e.relatedTarget || !this.menu.contains(e.relatedTarget)) return;

                                    let relatedTarget = e.relatedTarget;

                                    while(this.menu.contains(relatedTarget)){
                                        if(relatedTarget === this.currentElement) return;

                                        relatedTarget = relatedTarget.parentNode;
                                    }


                                    this.currentElement = null;

                                }
                            }
                        };

                        Hover.activate('.menu', '.column-container');

                    </script>
